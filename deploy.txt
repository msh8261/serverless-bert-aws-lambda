# Create an ECR repository
1- aws ecr create-repository --repository-name bert-lambda > /dev/null

# Register docker to ECR
2- aws ecr get-login-password --region us-east-2 | docker login --username AWS --password-stdin 095518905647.dkr.ecr.us-east-2.amazonaws.com


# tag / rename our previously created image to an ECR format
3- docker tag bert-lambda 095518905647.dkr.ecr.us-east-2.amazonaws.com/bert-lambda


# push the image to ECR Registry.
4- docker push 095518905647.dkr.ecr.us-east-2.amazonaws.com/bert-lambda


5- serverless plugin install -n serverless-python-requirements

6- serverless deploy
